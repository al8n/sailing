# Async storage (the M8-U1 fsync window): a submitted store write is visible but NOT durable until an
# explicit flush, so a node withholds its vote until the vote write is durable (persist-before-send).
# Three async voters: node 1 campaigns but casts no self-vote yet (its vote is only staged); the
# first flush makes node 1's vote durable and releases the RequestVotes; stabilize delivers them, and
# each follower stages its own vote WITHOUT replying; only after flushing 2 and 3 are their VoteResps
# released, letting node 1 reach quorum and become leader. (Full async commit propagation of the
# election no-op is exercised exhaustively by the VOPR; here we isolate the vote-durability gate.)
add-nodes 3 voters=(1,2,3) async=true
